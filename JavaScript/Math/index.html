# Math.random

// 随机数， 0 =< x < 1
// 得到两个数之间的随机整数
function getRandomInt(min, max) {
  min = Math.ceil(min);
  max = Math.floor(max);
  //不含最大值，含最小值
  return Math.floor(Math.random() * (max - min)) + min;
  //含最大值，含最小值 
  return Math.floor(Math.random() * (max - min + 1)) + min; 
}
