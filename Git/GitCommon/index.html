# 提交修改

// 查看状态
git status 

// 添加当前目录的所有文件的修改到暂存区
git add .

// 提交暂存区的内容到本地仓库
git commit -m "信息" 

// 推送内容到远程仓库，若没有该分支 则自动创建并推送
git push origin 分支名

// 强推代码（不建议）
git push -f
